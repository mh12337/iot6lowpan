## set up multi hop
# (hacker) commands to route through coordinator (8 -> 2 -> 4)
sudo ip -6 route add fd00::2 dev lowpan0
sudo ip -6 route add fd00::4 via fd00::2 dev lowpan0

### wpan-tools
# node join PAN
sudo iwpan dev wpan0 associate pan_id 0x04D2 coord <coordinator ext addr>
# send beacon from coord every 6 s
sudo iwpan dev wpan0 beacons send interval 6
# stop beacon 
sudo iwpan dev wpan0 beacons stop
# network associations
iwpan dev wpan0 list_associations

### iperf
# start server on port 5123
iperf3 -s -p 5123

# iperf client sends UDP at 50 kbps for 30 seconds
iperf3 -c <server-ip> -u -b 50k -t 30
# with pkg size
iperf3 -c fd00::2 -p 5123 -u -b 50k -t 30 -l 300
# remote
iperf3 -c ping.online.net -u -b 20k -l 80
# force ipv6 to remote server
iperf3 -6 -c ping.online.net -u -b 20k -l 80
